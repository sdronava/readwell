# ReadBook ŌĆö Implementing Scenario 4: Reader Opens a Book (Local Testing Plan)

## Overview

This document is the detailed implementation plan for **Scenario 4** from `specs/Phase1-MVP.md`:

```
1. Frontend: GET /api/v1/books
   ŌåÆ Gateway queries DB for published books, returns list.
2. User clicks "Read" on a book.
3. Frontend: GET /api/v1/books/{bookId}/metadata
   ŌåÆ Returns title, author, license, cover, total pages.
4. Frontend: GET /api/v1/books/{bookId}/pages/1
   ŌåÆ Returns page 1 structured content blocks (paragraphs, headings, code, images).
5. Frontend renders page from content blocks.
```

**Goal for local testing:** Run the entire read-a-book flow ŌĆö library view, book open, page navigation, and TTS ŌĆö entirely on localhost, using a book package already generated by the Document Conversion Service (i.e., the `books/` folder). No S3, no CloudFront, no PostgreSQL, no auth required.

## 3. Component 1 ŌĆö Local Static File Server (CDN Simulation)

**Port:** `9000`
**Serves:** the `books/` directory directly (each book package is a subdirectory)
**Purpose:** mimics CloudFront serving `{bookId}/assets/` and `{bookId}/pages/`

### Launch command


### URL shape

| Resource | URL |
|---|---|

**`local_store.py`** ŌĆö the core of the local simulation
```python
import json
from pathlib import Path
from content_gateway.config import settings

def list_books() -> list[dict]:
    """Scan BOOKS_DIR for subdirectories containing metadata.json."""
    root = Path(settings.books_dir)
    books = []
    for book_dir in sorted(root.iterdir()):
        meta_file = book_dir / "metadata.json"
        if book_dir.is_dir() and meta_file.exists():
            meta = json.loads(meta_file.read_text())
            books.append(_to_catalog_entry(book_dir.name, meta))
    return books

def get_metadata(book_id: str) -> dict:
    book_dir = Path(settings.books_dir) / book_id
    meta = json.loads((book_dir / "metadata.json").read_text())
    chapters = json.loads((book_dir / "chapters.json").read_text())
    cdn = f"{settings.content_base_url.rstrip('/')}/{book_id}"
    return {**meta, "cdnBaseUrl": cdn,
            "coverUrl": f"{cdn}/{meta.get('cover', '')}",
            "tableOfContents": chapters.get("chapters", [])}

def get_page(book_id: str, page_num: int) -> dict:
    path = Path(settings.books_dir) / book_id / "pages" / f"page_{page_num:03d}.json"
    if not path.exists():
        return None
    return json.loads(path.read_text())

def _to_catalog_entry(book_id: str, meta: dict) -> dict:
    cdn = f"{settings.content_base_url.rstrip('/')}/{book_id}"
    return {
        "bookId": book_id,
        "title": meta.get("title", "Unknown"),
        "author": meta.get("author", "Unknown"),
        "coverUrl": f"{cdn}/{meta.get('cover', '')}",
        "totalPages": meta.get("totalPages", 0),
        "description": meta.get("description"),
        "language": meta.get("language", "en"),
    }
```

**`routers/books.py`**
```python
from fastapi import APIRouter, HTTPException
from content_gateway.config import settings
from content_gateway import local_store, db_store

router = APIRouter(prefix="/api/v1")

def store():
    return local_store if settings.local_mode else db_store

@router.get("/books")
def list_books():
    return {"books": store().list_books()}

@router.get("/books/{book_id}/metadata")
def book_metadata(book_id: str):
    data = store().get_metadata(book_id)
    if data is None:
        raise HTTPException(404, f"Book not found: {book_id}")
    return data

@router.get("/books/{book_id}/chapters")
def book_chapters(book_id: str):
    data = store().get_metadata(book_id)
    return {"bookId": book_id, "chapters": data.get("tableOfContents", [])}

@router.get("/books/{book_id}/pages/{page_num}")
def book_page(book_id: str, page_num: int):
    page = store().get_page(book_id, page_num)
    if page is None:
        raise HTTPException(404, f"Page {page_num} not found in {book_id}")
    return page
```

### 5.4 Block rendering

The `Page` component iterates `blocks[]` and dispatches to typed renderers:

```tsx
function Page({ blocks, cdnBaseUrl }: { blocks: Block[]; cdnBaseUrl: string }) {
  return (
    <article>
      {blocks.map((block, i) => (
        <BlockRenderer key={i} block={block} cdnBaseUrl={cdnBaseUrl} />
      ))}
    </article>
  );
}

function BlockRenderer({ block, cdnBaseUrl }) {
  switch (block.type) {
    case "heading":
      const Tag = `h${block.level}` as keyof JSX.IntrinsicElements;
      return <Tag className="font-bold my-4">{block.text}</Tag>;

    case "paragraph":
      return <p className="my-3 leading-7">
        <EmphasisText text={block.text} emphasis={block.emphasis ?? []} />
      </p>;

    case "code":
      return <pre className="bg-gray-100 rounded p-4 overflow-x-auto my-4">
        <code className={block.language ? `language-${block.language}` : ""}>
          {block.text}
        </code>
      </pre>;

    case "list":
      const ListTag = block.ordered ? "ol" : "ul";
      return <ListTag className="list-disc pl-6 my-3">
        {block.items.map((item, j) => <li key={j}>{item}</li>)}
      </ListTag>;

    case "image":
      const src = block.srcset?.["800w"]
        ?? block.srcset?.["400w"]
        ?? block.srcset?.original
        ?? block.filename;
      return <figure className="my-6 text-center">
        <img
          src={`${cdnBaseUrl}/${src}`}
          alt={block.altText}
          srcSet={buildSrcSet(block.srcset, cdnBaseUrl)}
          className="mx-auto max-w-full"
        />
        {block.caption && <figcaption className="text-sm text-gray-500 mt-2">{block.caption}</figcaption>}
      </figure>;

    default:
      return null;
  }
}
```

> **Emphasis rendering:** `EmphasisText` takes the flat `text` string and the `emphasis[]` array (character ranges + style) and wraps the relevant ranges in `<strong>` or `<em>` tags.

### 5.5 TTS (Web Speech API)

The TTS controller:
1. Extracts readable text from `blocks[]` ŌĆö skips `code` blocks.
2. Calls `window.speechSynthesis.speak(utterance)`.
3. Attaches `boundary` event listener ŌåÆ highlights the current word in the rendered DOM.

```tsx
function useTTS(blocks: Block[]) {
  const speak = () => {
    window.speechSynthesis.cancel();
    const readable = blocks
      .filter(b => b.type === "paragraph" || b.type === "heading")
      .map(b => b.text)
      .join(" ");
    const utterance = new SpeechSynthesisUtterance(readable);
    utterance.addEventListener("boundary", (e) => {
      // highlight word at e.charIndex for e.charLength chars
    });
    window.speechSynthesis.speak(utterance);
  };
  return { speak, stop: () => window.speechSynthesis.cancel() };
}
```
